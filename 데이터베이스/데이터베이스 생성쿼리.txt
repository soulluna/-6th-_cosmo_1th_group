--부서 테이블
create table EMPLOYEE(
eno varchar2(20) not null,
pwd varchar2(20) not null,
ename varchar2(10) not null,
eng_name varchar2(20),
email varchar2(20) unique,
tel varchar2(11) unique,
dname varchar2(15) not null,
dname_two varchar2(8) not null,
hiredate date default sysdate,
rank varchar2(8),
isadmin varchar2(5) default 'false',
constraint employee_pk primary key(eno, ename, rank));

--전자결재페이지 table
create table APPROVAL(
applist varchar2(20),
txtnum number(10),
txtname varchar2(150) not null,
txtcont varchar2(3000) not null,
progress varchar2(10) not null,
entrydate date default sysdate,
middate date,
findate date,
eno varchar2(20) not null, 
ename varchar2(10) not null,
rank varchar2(8) not null,
midsugesteno varchar2(20),
finsugesteno varchar2(20),
vaclist varchar2(20),
vacstart date,
vacend date,
constraint APP_PK primary key(txtnum),
constraint APP_FK1 foreign key(eno,ename,rank) REFERENCES EMPLOYEE(eno,ename,rank)
);

--게시판 메인 table
create table NOTICE(
txtnum number(3),
txtname varchar2(100) not null,
txtcont varchar2(1000) not null,
entrydate date default sysdate,
eno varchar2(20) not null, 
ename varchar2(10) not null,
rank varchar2(8) not null,
noticelist number(1),
isAnnouncement char(1) default 'n',
likenum number(3) default 0,
viewtotal number(4) default 0,
constraint NOT_PK primary key(txtnum),
constraint NOT_FK1 foreign key(eno,ename,rank) REFERENCES EMPLOYEE(eno,ename, rank)

);

--스캐쥴러 테이블 추가
create table dailyscadul(
schnum number(3),
startdate timestamp,
enddate timestamp,
schname varchar2(100) not null,
schcont varchar2(500),
eno varchar2(20) not null, 
ename varchar2(10) not null,
rank varchar2(8) not null,
constraint sch_pk1 primary key(schnum),
constraint sch_fk1 foreign key(eno, ename, rank) references employee(eno, ename, rank)
);

--댓글 테이블 추가
create table commentTb(
    eno varchar2(20) not null, 
ename varchar2(10) not null,
rank varchar2(8) not null,
    txtnum number(3),
    comnum number(3),
    comcont varchar2(500),
    comdate date default sysdate,
    constraint COM_PK primary key(comnum),
    constraint COM_FK1 foreign key(eno,ename,rank) references employee(eno,ename,rank),
    constraint COM_FK2 foreign key(txtnum) references notice(txtnum)
);
